def load_images(filename):
  img = load_img(filename, color_mode="grayscale", target_size=(28,28))
  img = img_to_array(img)
  img = img.reshape(1,28,28,1)
  img = img.astype('float32')
  img = img / 255.0
  return img

model3 = models.Sequential()
